        for Index, Player in pairs(game:GetService("Players"):GetPlayers()) do

-- Function to track indivdual tools:
local TrackedTools = {}
local function TrackTool(Tool)
        pcall(function()
        
        local ToolAlreadyTracked = ((table.find(TrackedTools, Tool) and true) or false)
        if Tool ~= nil and not ToolAlreadyTracked then -- If the Tool exists and is not already being tracked then:
                table.insert(TrackedTools, #TrackedTools + 1, Tool) -- Insert the tool into the TrackedTools table
                print("Tracking "..Tool.Name)

                -- This allows you to find anyone who tries to bring you:
                -- (Either gets Character or Player's name depending on the parent of the tool)
                local ToolOriginalOwner = (Tool.Parent:IsA("Model") and Tool.Parent) or Tool.Parent.Parent

                -- Detect when the Tool's Parent has Changed:
                Tool:GetPropertyChangedSignal("Parent"):Connect(function()
                        -- If the tool is Parented to Local Character then:
                        if Tool ~= nil and Tool.Parent == game.Players.LocalPlayer.Character then
                                local LocalCharacter =
game.Players.LocalPlayer.Character

                                SavePosition()
                                warn(ToolOriginalOwner.Name.." tried to bring you!")

                                local RunStartTime = tick()
                                game:GetService("RunService"):BindToRenderStep("Anti-Bring", 1, function()
                                        if _G.AntiBring == true then
                                             if Tool ~= nil then
                                                    Tool.Parent = game.Players.LocalPlayer.Backpack
                                                    Tool:Destroy()
                                                end
                                                -- Engages No-Clip (Helps to stop the character from getting moved)
                                                game.Players.LocalPlayer.Character:FindFirstChildOfClass("Humanoid"):ChangeState(11)

                                                -- Stops the RunServiceBind after 0.075 Seconds
                                                if not Tool or (tick() - RunStartTime > 0.1 and Tool.Parent ~= game.Players.LocalPlayer.Character) then
                                                        game:GetService("RunService"):UnbindFromRenderStep("Anti-Bring")

                                                        wait()
                                                        LoadPosition(); wait(); LoadPosition()
                                                end
                                        end
                                end)
                        end
                end)
                end
                end)
end

local function ScanForTools(Player)
        spawn(function()
                wait()
                Player:WaitForChild("Backpack", 99e99)
                repeat wait() until Player.Character
                if Player ~= game.Players.LocalPlayer then
  
-- Scan for Tools in Backpack
                        for Index, Item in pairs(Player.Backpack:GetChildren()) do
                                if Item ~= nil and Item:IsA("Tool") then
                                        TrackTool(Item)
                                end
                        end
                        -- Scan for Tools in Character
                        for Index, Item in pairs(Player.Character:GetChildren()) do
                                if Item ~= nil and Item:IsA("Tool") then
                                        TrackTool(Item)
                                end
                        end
                        -- Scan for Added Tools in Backpack
                        Player.Backpack.ChildAdded:Connect(function(Item)
                                if Item ~= nil and Item:IsA("Tool") then
                                        TrackTool(Item)
                                end
                        end)
                        -- Scan for Added Tools in Character
                        Player.Character.ChildAdded:Connect(function(Item)
                                if Item ~= nil and Item:IsA("Tool") then
                                        TrackTool(Item)
                                end
                        end)

                        -- Scan for Added Tools in AddedCharacter
                        Player.CharacterAdded:Connect(function(Character)
                                -- Scan for Added Tools in Backpack
                                Player.Backpack.ChildAdded:Connect(function(Item)
                                        if Item ~= nil and Item:IsA("Tool") then
                                                TrackTool(Item)

            end
        end
    end

function dragify(Frame)
	dragToggle = nil
	dragSpeed = .1 -- You can edit this.
	dragInput = nil
	dragStart = nil
	dragPos = nil
	
	function updateInput(input)
		Delta = input.Position - dragStart
		Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + Delta.X, startPos.Y.Scale, startPos.Y.Offset + Delta.Y)
		game:GetService("TweenService"):Create(Frame, TweenInfo.new(dragSpeed), {Position = Position}):Play()
	end
	
	Frame.InputBegan:Connect(function(input)
		if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then
			dragToggle = true
			dragStart = input.Position
			startPos = Frame.Position
			input.Changed:Connect(function()
				if (input.UserInputState == Enum.UserInputState.End) then
					dragToggle = false
				end
			end)
		end
	end)
	
	Frame.InputChanged:Connect(function(input)
		if (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
			dragInput = input
		end
	end)
	
	game:GetService("UserInputService").InputChanged:Connect(function(input)
		if (input == dragInput and dragToggle) then
			updateInput(input)
		end
	end)
end

function getBoolFromString(str)
	local yes = {'true', 'yes', 'y', 'on'} 
	local no = {'false', 'no', 'n', 'off'}
	for i,v in pairs(yes) do yes[v] = true end 
	for i,v in pairs(no) do no[v] = true end 
	if yes[str] then 
		return true 
	elseif no[str] then 
		return false 
	else
		return nil 
	end
end 
local function GetPlayerFromString(nameString)
	if not nameString then return end 
	local matches= {}
	for _, player in ipairs (game.Players:GetPlayers()) do
		if string.lower(player.Name):match(string.lower(nameString)) then
			table.insert(matches, player)
		end
	end
	if #matches== 1 then
		return matches[1]
	else
		return nil
	end
end

				m_h.CFrame = plr_h.CFrame * CFrame.new(0,0,1.3)

		end
	end

function loop:create(name,func,waiit,...)
    local args = {...}
    if data.loops[name] and data.loops[name].active ~= nil then return end
    data.loops[name] = { }
    data.loops[name].active = true
    data.loops[name].loop = create(function()

            if type(waiit) == "userdata" then

                    wait(waiit)
                else
                    wait()

    until false
end)

--command stuff
function cmd:create(name,alias,desc,func)
    if not name or not func then return end
    local name = name:lower()
    if data.commands[name] then return end
    data.commands[name] = { }
    data.commands[name].func = func
    data.commands[name].alias = alias and alias or { }
    data.commands[name].desc = desc and desc or "no description"
end

function cmd:find(cmd)
    cmd = tostring(cmd)
    local cmd = cmd:lower()
    if data.commands[cmd] then return cmd end
    for index,value in pairs(data.commands) do
        if data:find_value(value,cmd) and data.commands[index] then
            return index
        end
    end
end

function cmd:run(cmd,...)
    if not self:find(cmd) then return end
    data.commands[cmd].func(...)
end

function cmd:find_and(str,s)
    if s == #str then return end
    local split = { }
    local x,y = str:find("and",s)
    if not x then return end
    local xx,yy = str:find("and",y)
    if not xx then
        string.gsub(str:sub(y + 1),"%S+",function(l) table_insert(split,l) end)
        local command = self:find(split[1])
        if command and command ~= "cmd" then
        self:run(command,select(2,unpack(split)))
    end
        return
    end
    string.gsub(str:sub(y + 1),"%S+",function(l) table_insert(split,l) end)
    local command = self:find(split[1])
    if self:find(command) and self:find(command) ~= "cmd" then
    self:run(command,select(2,unpack(split)))
    end
    return self:find_and(str,xx)
end

function cmd:log(command,...)
    warn("Command: "..tostring(command))
    table_foreach({...},print)
end

function cmd.exec(msg)
    local self = cmd
    if msg:lower():sub(1,#data.prefix) == data.prefix:lower() then
        local split = { }
        msg:gsub("%S+",function(w) table_insert(split,w) end)
        local command = self:find(split[1]:sub(#data.prefix+1))
        if command then
            self:run(command,select(2,unpack(split)))
            self:find_and(msg,1)
            cmd:log(command,select(2,unpack(split)))
        end
    end
end

cmd:create("kill2",{"k2"},"kills the user using the melee event.",function(...)
    local args = {...}
    local users = data:get_user(args[1])

    if type(users) ~= "table" then return end
    if #users == 0 then return end
    for index,player in pairs(users) do
        if player == client.player and data.me2 then client.humanoid.Health = 0 end
        for i=1,20 do
          melee_remote:FireServer(player)
        end
    end
end)

cmd:create("cmd",{},"adds fast command to fast commands table.",function(...)
    local args = {...}
    local what = args[1]
    local name = args[2]

    if what == "add" then
    if data.fast_commands[name] then return end
    data.fast_commands[name] = data.prefix..table.concat({select(3,unpack(args))}," ")
    return end
    if what == "run" then
    if not data.fast_commands[name] then return end
    cmd.exec(data.fast_commands[name])
    return end
end)
